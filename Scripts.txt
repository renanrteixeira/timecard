  `status` int(1) NOT NULL
) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci;;

CREATE TABLE `roles` ( 
		`id` INT NOT NULL AUTO_INCREMENT , 
		`name` VARCHAR(50) NOT NULL ,

CREATE TABLE `employees` ( 
		`id` INT NOT NULL AUTO_INCREMENT ,

		`rolefk` INT NOT NULL , 
		`birth` DATE NOT NULL , 
		`gender` VARCHAR(1) NOT NULL ,
